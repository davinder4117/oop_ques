Q11. C++ Program to convert Binary to Decimal



#include 
#include 
using namespace std;

int binaryToDecimal(long long binary) {
    int decimal = 0;
    int i = 0;

    while (binary != 0) {
        int remainder = binary % 10;
        decimal += remainder * pow(2, i);
        binary /= 10;
        i++;
    }

    return decimal;
}

int main() {
    long long binary;
    cout << "Enter binary number: ";
    cin >> binary;

    int decimal = binaryToDecimal(binary);

    cout << "Binary: " << binary << endl;
    cout << "Decimal: " << decimal << endl;

    return 0;
}

void decimalToBinary(int n) {
    int binary[32];
    int i = 0;

    while (n > 0) {
        binary[i] = n % 2;
        n = n / 2;
        i++;
    }

    cout << "Binary: ";
    for (int j = i - 1; j >= 0; j--) {
        cout << binary[j];
    }
    cout << endl;
}

    cout << "Decimal: " << decimal << endl;
    decimalToBinary(decimal);

    return 0;
}



Q13. C++ Program to convert Hex to Decimal



#include 
#include 
#include 
using namespace std;

int hexCharToInt(char ch) {
    if (ch >= '0' && ch <= '9')
        return ch - '0';
    else if (ch >= 'A' && ch <= 'F')
        return ch - 'A' + 10;
    else if (ch >= 'a' && ch <= 'f')
        return ch - 'a' + 10;
    return 0;
}

int hexToDecimal(char hex[]) {
    int decimal = 0;
    int len = strlen(hex);
    int base = 1;

    for (int i = len - 1; i >= 0; i--) {
        decimal += hexCharToInt(hex[i]) * base;
        base *= 16;
    }

    return decimal;
}

int main() {
    char hex[20];
    cout << "Enter hexadecimal number: ";
    cin >> hex;

    int decimal = hexToDecimal(hex);

    cout << "Hexadecimal: " << hex << endl;
    cout << "Decimal: " << decimal << endl;

    return 0;
}

#include 
using namespace std;

void decimalToHex(int n) {
    char hex[20];
    int i = 0;

    while (n != 0) {
        int remainder = n % 16;

        if (remainder < 10)
            hex[i++] = remainder + '0';
        else
            hex[i++] = remainder - 10 + 'A';

        n = n / 16;
    }

    cout << "Hexadecimal: ";
    for (int j = i - 1; j >= 0; j--) {
        cout << hex[j];
    }
    cout << endl;
}

int main() {
    int decimal;
    cout << "Enter decimal number: ";
    cin >> decimal;

    cout << "Decimal: " << decimal << endl;
    decimalToHex(decimal);

    return 0;
}
